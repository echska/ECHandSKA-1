    const galleryGrid = document.getElementById("galleryGrid");
    const videosCount = document.getElementById("videosCount");

    const modal = document.getElementById("videoModal");
    const modalVideo = document.getElementById("modalVideo");
    const modalSource = document.getElementById("modalSource");
    const modalTitle = document.getElementById("modalTitle");
    const modalDate = document.getElementById("modalDate");
    const modalFile = document.getElementById("modalFile");

    const closeModalBtn = document.getElementById("closeModalBtn");
    const prevBtn = document.getElementById("prevBtn");
    const nextBtn = document.getElementById("nextBtn");
    const stopBtn = document.getElementById("stopBtn");

    let currentIndex = 0;

    function getExt(filename){
      return filename.split(".").pop().toLowerCase();
    }

    function getType(filename){
      const ext = getExt(filename);
      if(ext === "mp4") return "video/mp4";
      if(ext === "mov") return "video/quicktime";
      return "video/mp4";
    }

    function openModal(index){
      currentIndex = index;
      const item = videosData[index];

      modalSource.src = item.file;
      modalSource.type = getType(item.file);
      modalVideo.load();

      modalTitle.textContent = item.title;
      modalDate.textContent = "Kalbin içinden geçen bir cümle";
      modalFile.textContent = item.quote || "";

      modal.classList.add("active");
      document.body.style.overflow = "hidden";
      modalVideo.play().catch(() => {});
    }

    function closeModal(){
      modalVideo.pause();
      modalSource.src = "";
      modalVideo.load();
      modal.classList.remove("active");
      document.body.style.overflow = "";
    }

    function prevVideo(){
      currentIndex = (currentIndex - 1 + videosData.length) % videosData.length;
      openModal(currentIndex);
    }

    function nextVideo(){
      currentIndex = (currentIndex + 1) % videosData.length;
      openModal(currentIndex);
    }

    function renderVideos(){
      videosCount.textContent = videosData.length;

      if(!videosData.length){
        galleryGrid.innerHTML = `<div class="empty">Şu anda video yok.</div>`;
        return;
      }

      galleryGrid.innerHTML = "";

      videosData.forEach((item, index) => {
        const card = document.createElement("article");
        card.className = "card";

        card.innerHTML = `
          <div class="thumb">
            <video src="${item.file}" muted preload="metadata" playsinline></video>
            <div class="play">▶</div>
          </div>
          <div class="card-body">
            <div class="date">Hatıra ${index + 1}</div>
            <div class="title">${item.title}</div>
            <div class="quote">${item.quote || ""}</div>
          </div>
        `;

        card.addEventListener("click", () => openModal(index));
        galleryGrid.appendChild(card);
      });
    }

    closeModalBtn.addEventListener("click", closeModal);
    stopBtn.addEventListener("click", () => {
      modalVideo.pause();
      modalVideo.currentTime = 0;
    });
    prevBtn.addEventListener("click", prevVideo);
    nextBtn.addEventListener("click", nextVideo);

    modal.addEventListener("click", (e) => {
      if(e.target === modal) closeModal();
    });

    document.addEventListener("keydown", (e) => {
      if(!modal.classList.contains("active")) return;

      if(e.key === "Escape") closeModal();
      if(e.key === "ArrowRight") prevVideo();
      if(e.key === "ArrowLeft") nextVideo();
    });

    renderVideos();
  </script>

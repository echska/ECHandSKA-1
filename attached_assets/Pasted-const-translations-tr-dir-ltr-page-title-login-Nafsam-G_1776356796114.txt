function getLang() {
  return localStorage.getItem("site_lang") || "tr";
}

function setLang(lang) {
  localStorage.setItem("site_lang", lang);
  applyTranslations(lang);
  updateAllCountdowns();
}

function applyTranslations(lang) {
  const t = translations[lang] || translations.tr;
  document.documentElement.lang = lang;
  document.documentElement.dir = t.dir;

  const isLogin = document.body.dataset.page === "login";
  document.title = isLogin ? t.page_title_login : t.page_title_main;

  document.querySelectorAll("[data-i18n]").forEach(el => {
    const key = el.dataset.i18n;
    if (t[key] !== undefined) el.textContent = t[key];
  });

  document.querySelectorAll("[data-i18n-placeholder]").forEach(el => {
    const key = el.dataset.i18nPlaceholder;
    if (t[key] !== undefined) el.setAttribute("placeholder", t[key]);
  });

  document.querySelectorAll(".lang-btn").forEach(btn => {
    btn.classList.toggle("active", btn.dataset.lang === lang);
  });

  if (counterBox && !isLogin) updateCounter();
}

document.addEventListener("click", (e) => {
  if (e.target.classList.contains("lang-btn")) {
    setLang(e.target.dataset.lang);
  }
});

document.querySelectorAll(".rain").forEach((rainEl) => {
  if (!rainEl.dataset.ready) {
    rainEl.dataset.ready = "1";
    for (let i = 0; i < 110; i++) {
      const drop = document.createElement("div");
      drop.className = "drop";
      drop.style.left = Math.random() * 100 + "vw";
      drop.style.animationDuration = (0.7 + Math.random() * 0.9) + "s";
      drop.style.animationDelay = Math.random() * 2 + "s";
      drop.style.opacity = 0.08 + Math.random() * 0.18;
      rainEl.appendChild(drop);
    }
  }
});

const typedEl = document.getElementById("typedText");
if (typedEl) {
  let phraseIndex = 0;
  let charIndex = 0;
  let deleting = false;

  function typeEffect() {
    const t = translations[getLang()] || translations.tr;
    const phrases = [t.typed_1, t.typed_2, t.typed_3, t.typed_4];
    const current = phrases[phraseIndex];

    if (!deleting) {
      typedEl.textContent = current.substring(0, charIndex + 1);
      charIndex++;
      if (charIndex === current.length) {
        deleting = true;
        setTimeout(typeEffect, 1200);
        return;
      }
    } else {
      typedEl.textContent = current.substring(0, charIndex - 1);
      charIndex--;
      if (charIndex === 0) {
        deleting = false;
        phraseIndex = (phraseIndex + 1) % phrases.length;
      }
    }

    setTimeout(typeEffect, deleting ? 40 : 80);
  }

  typeEffect();
}

const counterBox = document.getElementById("counterBox");
function updateCounter() {
  if (!counterBox) return;
  const startDate = new Date("2025-08-20T04:04:00");
  const now = new Date();
  const diff = now - startDate;
  const totalMinutes = Math.floor(diff / (1000 * 60));
  const days = Math.floor(diff / (1000 * 60 * 60 * 24));
  const hours = Math.floor((diff / (1000 * 60 * 60)) % 24);
  const minutes = totalMinutes % 60;
  const t = translations[getLang()] || translations.tr;

  counterBox.innerHTML =
    `${t.hero_eyebrow}<br><strong>${days}</strong> ${t.countdown_day} • <strong>${hours}</strong> ${t.countdown_hour} • <strong>${minutes}</strong> ${t.countdown_minute}`;
}
if (counterBox) {
  updateCounter();
  setInterval(updateCounter, 60000);
}

const redirectPage = "index.html";

function addMonthsAndDays(startDate, months, days) {
  const result = new Date(startDate);
  result.setMonth(result.getMonth() + months);
  result.setDate(result.getDate() + days);
  return result;
}

function formatRemaining(ms) {
  const totalSeconds = Math.floor(ms / 1000);
  const days = Math.floor(totalSeconds / (60 * 60 * 24));
  const hours = Math.floor((totalSeconds % (60 * 60 * 24)) / (60 * 60));
  const minutes = Math.floor((totalSeconds % (60 * 60)) / 60);
  const seconds = totalSeconds % 60;
  const t = translations[getLang()] || translations.tr;
  return `${days} ${t.countdown_day} • ${hours} ${t.countdown_hour} • ${minutes} ${t.countdown_minute} • ${seconds} ${t.countdown_second}`;
}

function updateAllCountdowns() {
  const first = document.getElementById("count-Ashkim");
  if (!first) return;

  const now = new Date();
  const startDate = new Date("2025-08-20T04:04:00");
  const endDate = addMonthsAndDays(startDate, 7, 26);
  const diff = endDate - now;
  const t = translations[getLang()] || translations.tr;

  [
    "count-Ashkim",
    "count-nafasim",
    "count-kaar",
    "count-asgoori",
    "count-lucifer",
    "count-ECHSKA"
  ].forEach(id => {
    const el = document.getElementById(id);
    if (!el) return;
    if (diff > 0) {
      el.className = "countdown-text";
      el.textContent = `${t.countdown_label}${formatRemaining(diff)}`;
    } else {
      el.className = "done-text";
      el.textContent = "• • • • •";
    }
  });
}

if (document.getElementById("count-Ashkim")) {
  updateAllCountdowns();
  setInterval(updateAllCountdowns, 1000);
}

function checkPassword() {
  if (!passwordInput || !msg) return;

  const value = passwordInput.value.trim().toLowerCase();
  const now = new Date();
  const startDate = new Date("2025-08-20T04:04:00");
  const endDate = addMonthsAndDays(startDate, 7, 26);
  const t = translations[getLang()] || translations.tr;

  if (now < endDate) {
    msg.style.color = "#fda4af";
    msg.textContent = t.login_msg_closed;
    return;
  }

  if (allowedPasswords.includes(value)) {
    msg.style.color = "#86efac";
    msg.textContent = t.login_msg_success;
    setTimeout(() => {
      window.location.href = redirectPage;
    }, 600);
  } else {
    msg.style.color = "#fda4af";
    msg.textContent = t.login_msg_wrong;
  }
}

if (openBtn) openBtn.addEventListener("click", checkPassword);
if (passwordInput) {
  passwordInput.addEventListener("keydown", function (e) {
    if (e.key === "Enter") checkPassword();
  });
}

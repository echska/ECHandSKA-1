    <script>
        const socket = io();
        let startTime = Date.now();

        // إرسال إحصائيات تلقائية كل 10 ثوانٍ
        setInterval(() => {
            const duration = Math.floor((Date.now() - startTime) / 1000);
            socket.emit('heartbeat-stats', {
                device: navigator.userAgent.includes('iPhone') ? 'iPhone 17 Pro Max' : 'جهاز غير معروف',
                status: duration > 30 ? 'تأمل عميق وتأثر' : 'تصفح سريع'
            });
        }, 10000);

        // استقبال أمر النبضة من السيرفر
        socket.on('trigger-haptic', () => {
            if (window.navigator && window.navigator.vibrate) {
                window.navigator.vibrate([100, 50, 100]);
            }
        });

        function sendPulse() { socket.emit('admin-pulse'); }
        function showAdmin() { 
            const pw = prompt("ادخل كلمة سر الزعيم:");
            if(pw === "UBOOR_2026") {
                document.getElementById('main-page').classList.add('hidden');
                document.getElementById('admin-panel').classList.remove('hidden');
            }
        }

        socket.on('update-dashboard', (stats) => {
            const body = document.getElementById('stats-body');
            body.innerHTML = stats.map(s => `
                <tr class="border-b border-zinc-800">
                    <td class="p-3">${s.time}</td>
                    <td class="p-3 text-blue-400">${s.device}</td>
                    <td class="p-3 text-green-400 font-bold">${s.status}</td>
                </tr>
            `).join('');
        });
    </script>
